local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local Button = workspace.WorkspaceCom["001_CanoeCloneButton"].Button
local ClickDetector = Button.ClickDetector
local Storage = workspace.WorkspaceCom["001_CanoeStorage"]
local SelectedPlayer = ""
local AbductionActive = false
local OriginalPos = nil
BangTab:AddSection({ "قائمة اللاعبين" })
local PlayerDropdown = BangTab:AddDropdown({
Name = "اختر الضحية",
Options = {},
Default = "",
Callback = function(Value)
SelectedPlayer = Value
end
})
local function RefreshPlayers()
local pList = {}
for _, p in pairs(Players:GetPlayers()) do
if p ~= LocalPlayer then table.insert(pList, p.Name) end
end
PlayerDropdown:SetOptions(pList)
end
BangTab:AddButton({
Name = "تحديث القائمة",
Callback = function()
RefreshPlayers()
end
})
BangTab:AddSection({ "نظام الخطف" })
BangTab:AddToggle({
Name = "خطف اللاعب المختار",
Default = false,
Callback = function(Value)
AbductionActive = Value
if Value then
if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
OriginalPos = LocalPlayer.Character.HumanoidRootPart.CFrame
end
end
end
})
task.spawn(function()
while true do
task.wait(0.1)
if AbductionActive and SelectedPlayer ~= "" then
local target = Players:FindFirstChild(SelectedPlayer)
if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") and target.Character:FindFirstChild("Humanoid") then
local char = LocalPlayer.Character
local hrp = char and char:FindFirstChild("HumanoidRootPart")
local hum = char and char:FindFirstChild("Humanoid")
if hrp and hum then
hrp.CFrame = Button.CFrame * CFrame.new(0, 3, 0)
task.wait(0.2)
fireclickdetector(ClickDetector)

repeat

for _, v in pairs(Storage:GetChildren()) do
if v.Name == "Canoe" then canoe = v break end
end
until canoe or (tick() - waitTime > 3) or not AbductionActive
if canoe then
local seat = canoe:FindFirstChildWhichIsA("VehicleSeat", true)
if seat then

local tRoot = target.Character.HumanoidRootPart

task.wait(0.1)
